150 template<typename Type, auto Data, typename Policy>
151 any getter([[maybe_unused]] handle handle, [[maybe_unused]] any index) {
152  auto dispatch = []([[maybe_unused]] auto &&value) {
153  if constexpr(std::is_same_v<Policy, as_void_t>) {
154  return any{std::in_place_type<void>};
155  } else if constexpr(std::is_same_v<Policy, as_alias_t>) {
156  return any{as_alias, std::forward<decltype(value)>(value)};
157  } else {
158  static_assert(std::is_same_v<Policy, as_is_t>);
159  return any{std::forward<decltype(value)>(value)};
160  }
161  };
162 
163  if constexpr(std::is_function_v<std::remove_pointer_t<decltype(Data)>> || std::is_member_function_pointer_v<decltype(Data)>) {
164  static_assert(std::is_invocable_v<decltype(Data), Type &>);
165  auto *clazz = handle.data<Type>();
166  return clazz ? dispatch(std::invoke(Data, *clazz)) : any{};
167  } else if constexpr(std::is_member_object_pointer_v<decltype(Data)>) {
168  using data_type = std::remove_cv_t<std::remove_reference_t<decltype(std::declval<Type>().*Data)>>;
169  static_assert(std::is_invocable_v<decltype(Data), Type *>);
170  auto *clazz = handle.data<Type>();
171 
172  if constexpr(std::is_array_v<data_type>) {
173  auto *idx = index.try_cast<std::size_t>();
174  return (clazz && idx) ? dispatch(std::invoke(Data, clazz)[*idx]) : any{};
175  } else {
176  return clazz ? dispatch(std::invoke(Data, clazz)) : any{};
177  }
178  } else {
179  static_assert(std::is_pointer_v<decltype(Data)>);
180 
181  if constexpr(std::is_array_v<std::remove_pointer_t<decltype(Data)>>) {
182  auto *idx = index.try_cast<std::size_t>();
183  return idx ? dispatch((*Data)[*idx]) : any{};

186  }
